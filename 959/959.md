## 959. 由斜杠划分区域
## 在由 1 x 1 方格组成的 N x N 网格 grid 中，每个 1 x 1 方块由 /、\ 或空格构成。这些字符会将方块划分为一些共边的区域。（请注意，反斜杠字符是转义的，因此 \ 用 "\\" 表示。）。返回区域的数目。

题目一共有三种输入 '\\', '/', ' '  

把每一个字符当作一个3*3的矩阵  

比如题目输入  

/\  
\\/  

将变成  
001100  
010010  
100001  
100001  
010010  
001100  
接下来数有多少个0块就可以了
用DFS或BFS实现皆可

code:
```
var dir = [][]int{{0, -1}, {0, 1}, {-1, 0}, {1, 0}}
var a = [100][100]int{}

func dfs(i, j, n int) {
	if a[i][j] == 1 {
		return
	}
	a[i][j] = 1
	for _, v := range dir {
		x := i + v[0]
		y := j + v[1]
		if x >= 0 && x < n && y >= 0 && y < n {
			dfs(i+v[0], j+v[1], n)
		}
	}
}

func regionsBySlashes(grid []string) int {
	n := len(grid)
	for i := 0; i <= n*3; i++ {
		for j := 0; j <= n*3; j++ {
			a[i][j] = 0
		}
	}

	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			x := i * 3
			y := j * 3
			switch grid[i][j] {
			case '/':
				a[x][y+2] = 1
				a[x+1][y+1] = 1
				a[x+2][y] = 1
			case '\\':
				a[x][y] = 1
				a[x+1][y+1] = 1
				a[x+2][y+2] = 1
			}
		}
	}

	ans := 0
	for i := 0; i < 3*n; i++ {
		for j := 0; j < 3*n; j++ {
			if a[i][j] == 0 {
				ans++
				dfs(i, j, n*3)
			}
		}
	}

	return ans
}
```
